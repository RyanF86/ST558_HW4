article_finder <- function(subject, 
                           date = format(Sys.Date() - 30, "%Y-%m-%d"), 
                           key = "173ec253cb06408abbf28061323e5fb4") {
  URL <- paste0("https://newsapi.org/v2/everything?q=",
                 subject,
                 "&from=",
                 date,
                 "&language=en&pageSize=50&apiKey=",
                 key)
  response <- httr::GET(URL)
  
  parsed_response <- fromJSON(rawToChar(response$content))
}
